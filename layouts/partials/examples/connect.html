package main
  
import (
  "os"
  
  "github.com/golungo/lungo"
)
  
func main() {
  uri := os.GetEnv("MONGO_URI")
  if err := lungo.Init(uri); err != nil {
    panic(err)
  }
  
  if err := lungo.Connect(); err != nil {
    panic(err)
  }
  
  defer func() {
    if err := lungo.Disconnect(); err != nil {
      panic(err)
    }
  }()
}
